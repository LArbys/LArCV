255  bool ProcessDriver::_process_entry_()
256  {
257  // Private method to execute processes and change entry number record
258  // This method does not perform any sanity check, hence private and
259  // should be used by wrapper method which performs necessary checks.
260 
261  // Execute
262  bool good_status=true;
263  bool cleared=false;
264  for(auto& p : _proc_v) {
265  good_status = good_status && p->_process_(_io);
266  if(!good_status && _enable_filter) break;
267  }
268  // No event-write to be done if _has_event_creator is set. Otherwise go ahead
269  if(!_has_event_creator) {
270  // If not read mode save entry
271  if(_io.io_mode() != IOManager::kREAD && (!_enable_filter || good_status)) {
272  cleared = true;
273  _io.save_entry();
274  }
275  if(!cleared)
276  _io.clear_entry();
277  cleared=true;
278  }
279  if(!cleared && _io.io_mode() == IOManager::kREAD)
280  _io.clear_entry();
281 
282  // Bump up entry record
283  ++_current_entry;
284 
285  return good_status;

339  void ProcessDriver::batch_process(size_t start_entry,size_t num_entries){
340  LARCV_DEBUG() << "Called" << std::endl;
341  // Public method to execute num_entries starting from start_entry
342  if(!start_entry) start_entry = _batch_start_entry;
343  if(!num_entries) num_entries = _batch_num_entry;
344 
345  // Check state
346  if(!_processing) {
347  LARCV_CRITICAL() << "Must call initialize() before start processing!" << std::endl;
348  throw larbys();
349  }
350 
351  // Define the max entry index for processing (1 above last entry to be processed)
352  size_t max_entry = start_entry + num_entries;
353 
354  // Check if start_entry is 0 for write mode (no entry should be specified for write mode)
355  if(_io.io_mode() == IOManager::kWRITE){
356  if(start_entry) {
357  LARCV_CRITICAL() << "Cannot specify start entry (1st arg) in kWRITE IO mode!" << std::endl;
358  throw larbys();
359  }
360  max_entry = _current_entry + num_entries;
361 
362  }else{
363  _current_entry = start_entry;
364  if(!num_entries) max_entry = start_entry + _io.get_n_entries();
365  }
366 
367  // Make sure max entry does not exceed the physical max from input. If so, truncate.
368  if(_io.io_mode() != IOManager::kWRITE && max_entry > _access_entry_v.size()){
369  LARCV_WARNING() << "Requested to process entries from " << start_entry << " to " << max_entry-1
370  << " ... but there are only " << _access_entry_v.size() << " entries in input!" << std::endl
371  << "Truncating the end entry to " << _access_entry_v.size()-1 << std::endl;
372  max_entry = _access_entry_v.size();
373  }
374 
375  // Batch-execute in while loop
376  size_t num_processed=0;
377  size_t num_fraction=(max_entry - _current_entry)/10;
378  while(_current_entry < max_entry) {
379 
380  if(_io.io_mode() != IOManager::kWRITE)
381 
382  _io.read_entry(_access_entry_v[_current_entry]);
383 
384  _process_entry_();
385 
386  ++num_processed;
387  if(!num_fraction) {
388  LARCV_NORMAL() << "Processed " << num_processed << " entries..." << std::endl;
389  }else if(num_processed%num_fraction==0) {
390  LARCV_NORMAL() << "Processed " << 10*int(num_processed/num_fraction) << " %..." << std::endl;
391  }
