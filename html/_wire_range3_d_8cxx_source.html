190  supera::GridPoint3D& max_pt) const {
191

193  WIRERANGE3D_CRITICAL() << "Configuration incomplete!" << std::endl;
194  throw std::exception();

197  static std::random_device rd;
198  static std::mt19937 mt(rd());
199 
200  auto const min_bound = _min_bound;
201  auto const max_bound = _max_bound;
202 
203  //
204  // Local algorithm parameters
205  // (note min_pt, max_pt are merely a container, not diagonal edge of containing cube!)
206  //
207  static const int MIN_DEFAULT = std::numeric_limits<int>::max();
208  static const int MAX_DEFAULT = std::numeric_limits<int>::min();
209  min_pt.x = min_pt.y = min_pt.z = MIN_DEFAULT;
210  max_pt.x = max_pt.y = max_pt.z = MAX_DEFAULT;
211 
212  //
213  // Determine x boundary
214  //
215  for(auto const& pt : points) {
216 
217  if( pt.x < min_bound.x || pt.x > max_bound.x ) {
218  WIRERANGE3D_DEBUG() << "Ignoring point Grid X = " << pt.x
219  << " as it is outside the bound: " << min_bound.x
220  << " => " << max_bound.x << std::endl;
221  continue;
222  }
223  WIRERANGE3D_DEBUG() << "Accepting point Grid X = " << pt.x << std::endl;
224 
225  if(pt.x < min_pt.x) min_pt.x = pt.x;
226  if(pt.x > max_pt.x) max_pt.x = pt.x;
227  }
228  if(min_pt.x == MIN_DEFAULT || max_pt.x == MAX_DEFAULT) {
229  WIRERANGE3D_INFO() << "No point found to fit within X constraint points! Generating random boundary within constraint..." << std::endl;
230  std::uniform_int_distribution<int> min_gen(min_bound.x,max_bound.x - _width.x + 1);
231  min_pt.x = min_gen(mt);
232  max_pt.x = min_pt.x + _width.x - 1;
233  WIRERANGE3D_INFO() << "New result: " << min_pt.x << " => " << max_pt.x << std::endl;
234  }else{
235  WIRERANGE3D_INFO() << "Generating X range..." << std::endl;
236  this->Randomize(min_pt.x, max_pt.x, min_bound.x, max_bound.x, _width.x, _padding.x);
237  }

239 
240  //
241  // Determine z boundary
242  //
243  for(auto const& pt : points) {
244

256  }

263  }else{
264  WIRERANGE3D_INFO() << "Generating Z range..." << std::endl;
265  this->Randomize(min_pt.z, max_pt.z, min_bound.z, max_bound.z, _width.z, _padding.z);
266  }

271  //
272  for(auto const& pt : points) {
273 
274  if( pt.x < min_pt.x || pt.x > max_pt.x)
275  continue;

301  void WireRange3D::Randomize(int& min_val, int& max_val,
302  const int min_bound, const int max_bound,
303  const int width, const int padding) const
304  {
305  WIRERANGE3D_INFO() << "Value range: " << min_val << " => " << max_val
306  << " ... absolute bound: " << min_bound << " => " << max_bound
307  << " ... target width: " << width << std::endl;
308  static std::random_device rd;
309  static std::mt19937 mt(rd());
310  int space = width - (max_val - min_val + 1);
311  int rand_bound_min = 0;
312  int rand_bound_max = 0;
313  if( space == 0 ) {
314  WIRERANGE3D_INFO() << "Range exactly same as the target width. Nothing to do..." << std::endl;
315  return;
316  }
317 
318  else if( space > 0 ) {
319  WIRERANGE3D_INFO() << "Data point range = " << (max_val - min_val + 1)
320  << " is smaller than target width = " << width << std::endl;
321  // Width found is smaller than limit.
322  // Randomize x-edge within the available space.
323  if( (min_val - min_bound) < (max_bound - max_val) ) {
324  rand_bound_min = std::max(-space, min_bound - min_val);
325  rand_bound_max = 0;
326  WIRERANGE3D_INFO() << "Closer to min bound. Generating uniform random: "
327  << rand_bound_min << " => " << rand_bound_max << std::endl;
328  std::uniform_int_distribution<int> shift_gen( rand_bound_min, rand_bound_max );
329  int shift = shift_gen(mt);
330  max_val = min_val + shift + width - 1;
331  min_val = min_val + shift;
332  WIRERANGE3D_INFO() << "Applying shift: " << shift << " ... result: " << min_val << " => " << max_val << std::endl;
333  }else{
334  rand_bound_min = 0;
335  rand_bound_max = std::min( space, max_bound - max_val);
336  WIRERANGE3D_INFO() << "Closer to max bound. Generating uniform random: "
337  << rand_bound_min << " => " << rand_bound_max << std::endl;
338  std::uniform_int_distribution<int> shift_gen( rand_bound_min, rand_bound_max );
339  int shift = shift_gen(mt);
340  min_val = max_val + shift - width + 1;
341  max_val = max_val + shift;
342  WIRERANGE3D_INFO() << "Applying shift: " << shift << " ... result: " << min_val << " => " << max_val << std::endl;
343  }
344 
345  }else {
346  WIRERANGE3D_INFO() << "Data point range = " << (max_val - min_val + 1)
347  << " is larger than target width = " << width << std::endl;
348  // Width found is larger than limit.
349  // Randomize x-edge within the available space.
350  rand_bound_min = min_val;
351  rand_bound_max = max_val - width + 1;
352  WIRERANGE3D_INFO() << "Generating uniform random: " << rand_bound_min << " => " << rand_bound_max << std::endl;
353  std::uniform_int_distribution<int> min_gen( rand_bound_min, rand_bound_max );
354  min_val = min_gen(mt);
355  max_val = min_val + width - 1;
356  WIRERANGE3D_INFO() << "New result: " << min_val << " => " << max_val << std::endl;
357  }
358  }
