239  std::uniform_int_distribution<> dis(0,_io.get_n_entries());
240

244  else if(entry >= _io.get_n_entries()) entry -= _io.get_n_entries();
245 
246  _io.read_entry(entry);

251  good_status = good_status && p->_process_(_io);
252  if(!good_status && _enable_filter) break;
253  }
254  if(!good_status && _enable_filter) continue;
255 
256  good_status = _filler->process(_io);
257  if(!good_status) continue;
258 
259  _batch_entries[valid_ctr] = entry;
260  ++valid_ctr;
261  }
