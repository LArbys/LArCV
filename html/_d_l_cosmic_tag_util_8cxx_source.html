123  std::vector< larcv::Image2D > DLCosmicTagUtil::makeIntimeCroppedImage( const std::vector<larcv::Image2D>& adc_wholeview_v, const int padding ) const {
124 
125  // check proper condition of class
126  if ( !fEntryLoaded )
127  LARCV_ERROR() << "Entry has not been loaded yet." << std::endl;
128 
129  // the numbers of planes should match
130  if ( adc_wholeview_v.size()!=m_pixelmask_vv.size() ) {
131  LARCV_ERROR() << "Number of input ADC plane images (" << adc_wholeview_v.size() << ") "
132  << "does not match with the number of pixelmask planes (" << m_pixelmask_vv.size() << ")"
133  << std::endl;
134  }
135 
136  // the output vector
137  std::vector< larcv::Image2D > img_v;
138 
139  //return empty vector if no clusters
140  if ( m_larflowcluster_v->size()==0 )
141  return img_v;
142 
143  // we get the bounding boxes from each of the pixelmasks and form a axis-aligned union bounding box
144  // the bounding boxes are represented by the image2d imagemeta class
145  std::vector< larcv::ImageMeta > meta_v;
146 
147  // setup the first meta(s)
148  size_t plane = 0;
149  for ( auto const& pevent_pixelmask : m_pixelmask_vv ) {
150  auto const& mask = pevent_pixelmask->front();
151  larcv::ImageMeta meta = DLCosmicTagUtil::metaFromPixelMask( mask, plane );
152  meta_v.emplace_back( std::move(meta) );
153  }
154 
155  // now loop through the rest of the masks, updating the imagemeta
156  for ( plane=0; plane<m_pixelmask_vv.size(); plane++ ) {
157  size_t nclusters = m_pixelmask_vv[plane]->size();
158  for ( size_t icluster=1; icluster<nclusters; icluster++ ) {
159  auto const& mask = (*m_pixelmask_vv[plane])[icluster];
160  larcv::ImageMeta cluster_meta = DLCosmicTagUtil::metaFromPixelMask( mask, plane );
161  larcv::ImageMeta& union_meta = meta_v.at(plane);
162  union_meta = union_meta.inclusive( cluster_meta );
163  }
164  }
165 
166  // now that we have union metas for each plane, crop out the regions
167  // if for some reason we extend outside the range, we make an intersection box
168  for ( plane=0; plane<m_pixelmask_vv.size(); plane++ ) {
169  larcv::ImageMeta& unionmeta = meta_v.at(plane);
170  const larcv::ImageMeta& adcmeta = adc_wholeview_v.at(plane).meta();
171  if ( !adcmeta.contains( unionmeta ) ) {
172  unionmeta = unionmeta.overlap( adcmeta );
173  }
174 
175  // make crop
176  larcv::Image2D crop = adc_wholeview_v.at(plane).crop( unionmeta );
177  img_v.emplace_back( std::move(crop) );
178  }
179 
180  return img_v;
181  }
