131  _cols = image_v.front().meta().cols();

143  _mirror_caffe_idx_to_img_idx.resize(_rows*_cols,0);
144  size_t caffe_idx = 0;
145  for(size_t row=0; row<_rows; ++row) {
146  for(size_t col=0; col<_cols; ++col) {
147  _caffe_idx_to_img_idx[caffe_idx] = col*_rows + row;
148  _mirror_caffe_idx_to_img_idx[caffe_idx] = (_cols-col-1)*_rows + row;
