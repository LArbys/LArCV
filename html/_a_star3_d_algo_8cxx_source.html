773  float AStar3DAlgo::distanceFromCentralLine( const std::vector<float>& start_tyz, const std::vector<float>& end_tyz, const std::vector<float>& testpt_tyz ) {
774  // returns the shortest distance of the test point from the line segment between the start and end points
775  // the coordinate system is in (tick, Y, Z)
776  // will return a value in cm
777  // stole formular from: http://mathworld.wolfram.com/Point-LineDistance3-Dimensional.html
778  float norm12 = 0;
779  std::vector<float> dir10( start_tyz.size(), 0.0 );
780  std::vector<float> dir20( start_tyz.size(), 0.0 );
781  std::vector<float> dir12( start_tyz.size(), 0.0 );
782  for (size_t i=0; i<start_tyz.size(); i++) {
783  dir10[i] = start_tyz.at(i) - testpt_tyz.at(i);
784  dir20[i] = end_tyz.at(i) - testpt_tyz.at(i);
785  dir12[i] = end_tyz.at(i) - start_tyz.at(i);
786  if (i==0) {
787  dir10[0] *= ::larutil::LArProperties::GetME()->DriftVelocity()*0.5; // [cm/usec]*[usec/tick]
788  dir20[0] *= ::larutil::LArProperties::GetME()->DriftVelocity()*0.5; // [cm/usec]*[usec/tick]
789  dir12[0] *= ::larutil::LArProperties::GetME()->DriftVelocity()*0.5; // [cm/usec]*[usec/tick]
790  }
791  norm12 += dir12[i]*dir12[i];
792  }
793  norm12 = sqrt(norm12);
794  if ( norm12==0 ) {
795  throw std::runtime_error("AStar3DAlgo::distanceFromCentralLine[error] start and end point are the same. calculation undefined.");
796  }
797 
798 
799  // dir01 x dir02
800  std::vector<float> dirX(dir10.size(),0.0);
801  dirX[0] = dir10[1]*dir20[2] - dir10[2]*dir20[1];
802  dirX[1] = dir10[2]*dir20[0] - dir10[0]*dir20[2];
803  dirX[2] = dir10[0]*dir20[1] - dir10[1]*dir20[0];
804  float normX = 0.;
805  for (size_t i=0; i<dirX.size(); i++) {
806  normX += dirX[i]*dirX[i];
807  }
808  normX = sqrt(normX);
809 
810  float dist = normX/norm12;
811  return dist;
812  }
