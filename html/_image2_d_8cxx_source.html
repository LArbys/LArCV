172  std::vector<float> Image2D::copy_compress(size_t rows, size_t cols, CompressionModes_t mode) const
173  {
174  if(mode == kOverWrite)
175  throw larbys("kOverWrite is invalid for copy_compress!");
176 
177  const size_t self_cols = _meta.cols();
178  const size_t self_rows = _meta.rows();
179  if(self_cols % cols || self_rows % rows) {
180  char oops[500];
181  sprintf(oops,"Compression only possible if height/width are modular 0 of compression factor! H:%zuMOD%zu=%zu W:%zuMOD%zu=%zu",
182  self_rows,rows,self_rows%rows,self_cols,cols,self_cols%cols);
183  throw larbys(oops);
184  }
185  size_t cols_factor = self_cols / cols;
186  size_t rows_factor = self_rows / rows;
187  std::vector<float> result(cols*rows,0);
188 
189  for(size_t col=0; col<cols; ++col) {
190  for(size_t row=0; row<rows; ++row) {
191  float value = 0;
192 
193  for(size_t orig_col=col*cols_factor; orig_col<(col+1)*cols_factor; ++orig_col)
194  for(size_t orig_row=row*rows_factor; orig_row<(row+1)*rows_factor; ++orig_row) {
195 
196  if ( mode!=kMaxPool ) {
197  // for sum and average mode
198  value += _img[orig_col * self_rows + orig_row];
199  }
200  else {
201  // maxpool
202  value = ( value<_img[orig_col * self_rows + orig_row] ) ? _img[orig_col * self_rows + orig_row] : value;
203  }
204  }
205 
206  result[col*rows + row] = value;
207  if ( mode==kAverage )
208  result[col*rows + row] /= (float)(rows_factor*cols_factor);
209  }
210  }
211  return result;
212  }

263  {

265 
266  if(rhs_meta.pixel_height() != _meta.pixel_height() || rhs_meta.pixel_width() != _meta.pixel_width())

269 
270  double x_min = std::max(_meta.min_x(),rhs_meta.min_x());
271  double x_max = std::min(_meta.max_x(),rhs_meta.max_x());
272  if(x_min >= x_max) return;
273 
274  double y_min = std::max(_meta.min_y(),rhs_meta.min_y());
275  double y_max = std::min(_meta.max_y(),rhs_meta.max_y());
276  if(y_min >= y_max) return;
277 
278  size_t row_min1 = _meta.row(y_max);
279  size_t col_min1 = _meta.col(x_min);
280 
281  size_t row_min2 = rhs_meta.row(y_max);
282  size_t col_min2 = rhs_meta.col(x_min);
283 
284  size_t nrows = (y_max - y_min) / _meta.pixel_height();
285  size_t ncols = (x_max - x_min) / _meta.pixel_width();
286 
287  auto const& img2 = rhs.as_vector();
288 
289  for(size_t col_index=0; col_index < ncols; ++col_index) {
290 
291  size_t index1 = _meta.index(row_min1,col_min1+col_index);
292  size_t index2 = rhs_meta.index(row_min2,col_min2+col_index);
293 
294  switch(mode) {
295 
296  case kSum:
297 
298  for(size_t row_index=0; row_index < nrows; ++row_index)
299 
300  _img[index1+row_index] += img2[index2+row_index];
301 
302  break;
303 
304  case kAverage:
305 
306  for(size_t row_index=0; row_index < nrows; ++row_index)
307 
308  _img[index1+row_index] += (_img[index1+row_index] + img2[index2+row_index]) / 2.;
309 
310  break;
311 
312  case kMaxPool:
313 
314  for(size_t row_index=0; row_index < nrows; ++row_index)
315 
316  _img[index1+row_index] = std::max(_img[index1+row_index],img2[index2+row_index]);
317 
318  break;
319 
320  case kOverWrite:
321 
322  for(size_t row_index=0; row_index < nrows; ++row_index)
323 
324  _img[index1+row_index] = img2[index2+row_index];
