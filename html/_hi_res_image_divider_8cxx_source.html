301  void HiResImageDivider::generateSingleCosmicDivision( std::vector< int >& divlist, const EventImage2D& input_event_images, larcv::ROI& roi ) {
302  // we randomly pick a division (by randomly drawing a position in the detector
303  // we return a division which satisfies some minimum amount of interesting pixel threshold
304  bool viewok = false;
305  int ntries = 0;
306 
307  roi.Type(kROICosmic);
308 
309  int idiv = -1;
310 
311  while (!viewok && ntries<fMaxRedrawAttempts) {
312  ntries++;
313  // FIXME: need a way to get detector dimension somehow...
314  const double zmin = 0;
315  const double zmax = 1036.0;
316  const double ymin = -116.;
317  const double ymax = 116.;
318  const double xmin = 0.;
319  const double xmax = 255.;
320  const double tmin = 3125.; // in ns
321  const double tmax = tmin + 1600.;
322  std::random_device rd;
323  std::mt19937 gen(rd());
324  std::uniform_real_distribution<> dis(0.,1.);
325  roi.Position( dis(gen) * (xmax - xmin) + xmin,
326  dis(gen) * (ymax - ymin) + ymin,
327  dis(gen) * (zmax - zmin) + zmin,
328  dis(gen) * (tmax - tmin) + tmin);
329 
330  idiv = findVertexDivision( roi );
331  if ( idiv==-1 ) {
332  viewok = false;
333  continue;
334  }
335  larcv::hires::DivisionDef const& vertex_div = m_divisions.at( idiv );
336  larcv::EventImage2D cosmic_test;
337  cropEventImages( input_event_images, vertex_div, cosmic_test );
338  if ( isAbovePixelThreshold( cosmic_test ) ) {
339  viewok = true;
340  break;
341  }
342  }
343  if ( !viewok ) {
344  LARCV_ERROR() << "could not find cosmic roi with enough interesting pixels.\n" << std::endl;
345  }
346  else {
347  divlist.push_back(idiv);
348  }
349  }

512

517  int **planebounds = new int*[fNPlanes];

529  }
530 
531  float zbounds[2];
532  float xbounds[2];
533  float ybounds[2];
534  int tickbounds[2];
535 
536  t->SetBranchAddress( "zbounds", zbounds );
537  t->SetBranchAddress( "ybounds", ybounds );
538  t->SetBranchAddress( "xbounds", xbounds );
539  t->SetBranchAddress( "tickbounds", tickbounds );
540 
541  fMaxWireInRegion = 0;

548  }

552  plane1[i] = (int)planebounds[1][i];
553  plane2[i] = (int)planebounds[2][i];
554  tickbounds[i] *= fTickPreCompression;
555  tickbounds[i] += fTickStart;
556  }
557  plane0[1] += fWirePreCompression;
558  plane1[1] += fWirePreCompression;

595  void HiResImageDivider::CalculateDivisions() {
596 
597  LARCV_INFO() << "Calculating Divisions." << std::endl;
598 
599  std::set<larcv::hires::DivisionID> unique_div_s;
600  fMaxWireImageWidth = 0; // will fill this later
601 
602  float zbounds[2] = { 0.0, 1037.0 }; //{ 0.04, 1036.45 };
603  float ybounds[2] = { -116.0, 118.0 }; //{-115.51,117.44};
604  float tbounds[2] = { 2400.0, 8448.0 }; // 2400+6048 // TICKS
605  float trigger_tick = 3200; // sets 'x=0'
606 
607  float zWidth = (zbounds[1]-zbounds[0])/fNumZdivisions;
608  float yWidth = (ybounds[1]-ybounds[0])/fNumYdivisions;
609  float tWidth = (tbounds[1]-tbounds[0])/fNumTdivisions; // in ticks
610 
611  int nz = fNumZdivisions;
612  int ny = fNumYdivisions;
613  int nt = fNumTdivisions;
614  float zstep = zWidth;
615  float ystep = yWidth;
616  float tstep = tWidth;
617  float driftx = 0.11*0.5; // (0.11 cm per microsecond) * 0.5 microseconds per tick
618 
619  if ( fOverlapDivisions ) {
620  nz = 2*fNumZdivisions-1;
621  ny = 2*fNumYdivisions-1;
622  nt = 2*fNumTdivisions-1;
623  zstep *= 0.5;
624  ystep *= 0.5;
625  tstep *= 0.5;
626  }
627 
628 
629  std::vector<int> uwires;
630  std::vector<int> vwires;
631  std::vector<int> ywires;
632 
633  divalgo::DividerAlgo algo;
634 
635  for (int iz=0; iz<nz; iz++) {
636  for (int iy=0; iy<ny; iy++) {
637 
638  float box_z[2] = { zbounds[0]+iz*zstep, zbounds[0]+iz*zstep+zWidth };
639  float box_y[2] = { ybounds[0]+iy*ystep, ybounds[0]+iy*ystep+zWidth };
640  float bbox[4] = { box_z[0], box_y[0], box_z[1], box_y[1] };
641  algo.getregionwires( bbox[0], bbox[1], bbox[2], bbox[3], ywires, uwires, vwires );
642 
643  int ubounds[2] = { uwires.at(0), uwires.back() };
644  int vbounds[2] = { vwires.at(0), vwires.back() };
645  int ybounds[2] = { ywires.at(0), ywires.back() };
646 
647  if ( (int)uwires.size()>fMaxWireImageWidth ) fMaxWireImageWidth = (int)uwires.size();
648  if ( (int)vwires.size()>fMaxWireImageWidth ) fMaxWireImageWidth = (int)vwires.size();
649  if ( (int)ywires.size()>fMaxWireImageWidth ) fMaxWireImageWidth = (int)ywires.size();
650 
651  for (int it=0; it<nt; it++) {
652 
653  int tickbounds[2] = { (int)(tbounds[0] + it*tstep), (int)(tbounds[0] + it*tstep + tWidth) };
654  float box_x[2] = { (tickbounds[0]-trigger_tick)*driftx, (tickbounds[1]-trigger_tick)*driftx };
655 
656  DivisionDef div( ubounds, vbounds, ybounds, tickbounds, box_x, box_y, box_z );
657 
658  auto const& div_id = div.ID();
659  if(unique_div_s.find(div_id) == unique_div_s.end()) {
660  unique_div_s.insert(div.ID());
661  m_divisions.emplace_back(div);
662  }
663 
664  }//end of t-loop
665 
666  }//end of y-loop
667  }//end of z-loop
668 
669  }
670 
671  }
