308  bool ThreadDatumFiller::_batch_process_(size_t nentries)
309  {
310  LARCV_DEBUG() << " start" << std::endl;
311  _thread_state = kThreadStateRunning;
312  _filler->_nentries = nentries;
313  if (!_processing) {
314  LARCV_INFO() << "Initializing for 1st time processing" << std::endl;
315  _driver.initialize();
316  _processing = true;
317  }
318 
319  size_t last_entry = kINVALID_SIZE - 1;
320  if (_batch_entries.size()) last_entry = _batch_entries.back();
321  _batch_entries.resize(nentries, 0);
322  _batch_events.clear();
323  _batch_events.reserve(nentries);
324 
325  _filler->_nentries = nentries;
326  _filler->batch_begin();
327 
328  size_t valid_ctr = 0;
329 
330  std::random_device rd;
331  std::mt19937 gen(rd());
332  std::uniform_int_distribution<> dis(0, _driver.io().get_n_entries() - 1);
333  if (_random_access)
334  LARCV_INFO() << "Generating random numbers from 0 to " << _driver.io().get_n_entries() << std::endl;
335 
336  // counter for _optional_next_index_v
337  size_t next_batch_ctr = 0;
338 
339  LARCV_INFO() << "Entering process loop" << std::endl;
340  while (valid_ctr < nentries) {
341  size_t entry = last_entry + 1;
342  if(_optional_next_index_v.size()) {
343  if(next_batch_ctr >= _optional_next_index_v.size())
344  break;
345  entry = _optional_next_index_v[next_batch_ctr];
346  ++next_batch_ctr;
347  }else{
348  if (_optional_next_index != kINVALID_SIZE) {
349  entry = _optional_next_index;
350  _optional_next_index = kINVALID_SIZE;
351  }
352  if (entry == kINVALID_SIZE) entry = 0;
353 
354  if (_random_access) {
355  entry = dis(gen);
356  while (entry == last_entry) entry = dis(gen);
357  }
358  else if (entry >= _driver.io().get_n_entries()) entry -= _driver.io().get_n_entries();
359 
360  }
361 
362  LARCV_INFO() << "Processing entry: " << entry
363  << " (tree index=" << _driver.get_tree_index( entry ) << ")" << std::endl;
364 
365  last_entry = entry;
366  bool good_status = _driver.process_entry(entry, true);
367  if (_enable_filter && !good_status) {
368  LARCV_INFO() << "Filter enabled: bad event found" << std::endl;
369  continue;
370  }
371  LARCV_INFO() << "Finished processing event id: " << _driver.event_id().event_key() << std::endl;
372 
373  _batch_entries[valid_ctr] = _driver.get_tree_index( entry );
374  _batch_events.push_back(_driver.event_id());
375  ++valid_ctr;
376  LARCV_INFO() << "Processed good event: valid entry counter = " << valid_ctr << " : " << _batch_events.size() << std::endl;
377  }
378  _num_processed += valid_ctr;
379  _filler->batch_end();
380  _thread_state = kThreadStateIdle;
381  _optional_next_index = kINVALID_SIZE;
382  _optional_next_index_v.clear();
383  LARCV_DEBUG() << " end" << std::endl;
384  return true;
385  }
