279  bool SuperaMCROI::FilterNode(const supera::MCNode& node) const
280  {
281  if(node.source_type == supera::MCNode::SourceType_t::kMCTruth) {
282  if(_pass_origin && node.origin != _pass_origin)
283  return false;
284  }else if(node.source_type == supera::MCNode::SourceType_t::kMCTrack) {
285  auto const& mctrack = LArData<supera::LArMCTrack_t>()[node.source_index];
286  LARCV_DEBUG() << "MCTrack InitE " << mctrack.Start().E() << " ... DepE "
287  << (mctrack.size() >1 ? mctrack.front().E() - mctrack.back().E() : 0) << std::endl;
288  if(mctrack.Start().E() < _track_min_einit) return false;
289  if(_track_min_edep > 0) {
290  if(mctrack.size()<2) return false;
291  if( (mctrack.front().E() - mctrack.back().E()) < _track_min_edep ) return false;
292  }
293  for(size_t filter_idx=0; filter_idx<_filter_pdg.size(); ++filter_idx) {
294  auto const& pdg = _filter_pdg[filter_idx];
295  if(pdg != mctrack.PdgCode()) continue;
296  auto const& filter_min_einit = _filter_min_einit[filter_idx];
297  if(mctrack.Start().E() < filter_min_einit) return false;
298  auto const& filter_min_edep = _filter_min_edep[filter_idx];
299  if(filter_min_edep > 0) {
300  if(mctrack.size()<2) return false;
301  if( (mctrack.front().E() - mctrack.back().E()) < filter_min_edep ) return false;
302  }
303  }
304  }else if(node.source_type == supera::MCNode::SourceType_t::kMCShower) {
305  auto const& mcshower = LArData<supera::LArMCShower_t>()[node.source_index];
306  LARCV_DEBUG() << "MCShower InitE " << mcshower.Start().E() << " ... DepE "
307  << mcshower.DetProfile().E() << std::endl;
308  if(mcshower.Start().E() < _shower_min_einit) return false;
309  if(mcshower.DetProfile().E() < _shower_min_edep) return false;
310  for(size_t filter_idx=0; filter_idx<_filter_pdg.size(); ++filter_idx) {
311  auto const& pdg = _filter_pdg[filter_idx];
312  if(pdg != mcshower.PdgCode()) continue;
313  auto const& filter_min_einit = _filter_min_einit[filter_idx];
314  if(mcshower.Start().E() < filter_min_einit) return false;
315  auto const& filter_min_edep = _filter_min_edep[filter_idx];
316  if( mcshower.DetProfile().E() < filter_min_edep) return false;
317  }
318  }
319  return true;
320  }

324  {
325  larcv::ROI res;
326  if(node.source_type == supera::MCNode::SourceType_t::kMCTruth)

345  auto const& part_meta = res.BB().at(plane);
